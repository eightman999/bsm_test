    id = LON.20
    title = LON.20.t
    desc = LON.20.desc
    picture = GFX_event_empty
    
    is_triggered_only = yes
    
    option = {
        name = LON.20.a # 投資を受け入れる
        ai_chance = {
            factor = 90
        }
        custom_effect_tooltip = LON_investment_accepted_tooltip
    }
    
    option = {
        name = LON.20.b # 投資を拒否する

            factor = 10
        }

        custom_effect_tooltip = LON_investment_rejected_tooltip
    }
}

# 民主主義国への投資イベント
country_event = {
    id = LON.21
    title = LON.21.t
    desc = LON.21.desc
    picture = GFX_event_empty
    
    is_triggered_only = yes
    
    option = {
        name = LON.21.a # 投資を歓迎する
        ai_chance = {
            factor = 85
        }
        custom_effect_tooltip = LON_democracy_investment_accepted_tooltip
    }
    
    option = {
        name = LON.21.b # 外国の干渉を拒否
        ai_chance = {
            factor = 15
            modifier = {
                factor = 0.3
                has_government = democratic_ideology
                democratic_ideology > 0.7
            }
        }
        FROM = {
            add_political_power = 150 # 投資コストを返還
        }
        add_stability = 0.02 # 独立性を保った満足感
        custom_effect_tooltip = LON_democracy_investment_rejected_tooltip
    }
}

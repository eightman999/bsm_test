    id = LON.leader_change.1
    title = LON.leader_change.1.t
    desc = LON.leader_change.1.desc
    
    is_triggered_only = yes
    major = yes
    
    immediate = {
        # 現在のリーダーを退任させ、次のリーダーを任命
        # LON_GBは既存のキャラクターで、この例ではHarry Twyfordが初期リーダー
        set_country_leader_ideology = neutrality_democracy #現在のリーダーのイデオロギーを設定（念のため）
        retire_country_leader = yes # 現在のリーダーを退任させる

            character = LON_frank_bowater
            promote_leader = yes
            country_leader = {
                ideology = neutrality_democracy
            }
        }
    }
    
    option = {
        name = EVT_OK
    }
}

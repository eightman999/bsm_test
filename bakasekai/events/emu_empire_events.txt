# エミューの大移住
country_event = {
	id = emu_empire.5
	title = emu_empire.5.t
	desc = emu_empire.5.d
	picture = GFX_report_event_generic_factory
	
	trigger = {
		tag = AUS
		has_completed_focus = AUS_feather_industry
		any_owned_state = {
			free_building_slots = {
				building = industrial_complex
				size > 1
			}
		}
	}
	
	fire_only_once = yes
	
	option = {
		name = emu_empire.5.a
		random_owned_controlled_state = {
			limit = {
				free_building_slots = {
					building = industrial_complex
					size > 1
				}
			}
			add_building_construction = {
				type = industrial_complex
				level = 1
				instant_build = yes
			}
		}
		ai_chance = { factor = 100 }
	}
	
	option = {
		name = emu_empire.5.b
		add_political_power = 100
		ai_chance = { factor = 0 }
	}
}
